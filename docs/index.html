Fold DOM elements without cloning the app

Pseudo-3D origami for real DOM.

Orikata renders folded visual panels, then maps pointer hits back to the original source element. KISS first: static views, baked manifests, button clicks, and text input proxies.

import { ROOT_ID, StaticImageSnapshotProvider, createOrigamiRuntime } from 'orikata';

const runtime = createOrigamiRuntime({
  mode: 'interactive-bridge',
  host,
  sourceRoot,
  paper: { width: 420, height: 220 },
  foldOps: [{
    id: 'fold-right',
    targetNodeId: ROOT_ID,
    childNodeId: 'right-panel',
    line: { a: { x: 210, y: 0 }, b: { x: 210, y: 220 } },
    movingSide: 1,
    angleDeg: -35
  }],
  snapshotProvider: new StaticImageSnapshotProvider(snapshot)
});

await runtime.mount();
